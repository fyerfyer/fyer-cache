Node-Sharded Caching

节点分片缓存是 FyerCache 分布式功能的核心组件，它通过一致性哈希算法将数据分布到多个缓存节点上，实现水平扩展、负载均衡和高可用性。

NodeShardedCache 架构​

NodeShardedCache 是 FyerCache 中实现节点分片缓存的核心类，其主要组件包括：

创建与配置​

基本创建​

创建一个节点分片缓存非常简单：

// 创建节点分片缓存
shardedCache := cache.NewNodeShardedCache()

// 添加三个缓存节点 
node1 := cache.NewMemoryCache()
node2 := cache.NewMemoryCache()
node3 := cache.NewMemoryCache()

// 添加节点到分片缓存，可以为不同节点指定不同权重
shardedCache.AddNode("node1", node1, 1)  // 权重为1
shardedCache.AddNode("node2", node2, 2)  // 权重为2，将接收约两倍的键
shardedCache.AddNode("node3", node3, 1)  // 权重为1

配置选项​

NodeShardedCache 支持多种配置选项：

// 创建带配置选项的节点分片缓存
shardedCache := cache.NewNodeShardedCache(
    // 配置每个物理节点的虚拟节点数量，影响数据分布均匀性
    cache.WithNodeHashReplicas(200),  
    
    // 配置副本因子，每个键存储在多少个节点上（用于容错）
    cache.WithNodeReplicaFactor(2),
    
    // 配置自定义哈希函数
    cache.WithNodeCustomHashFunc(myHashFunction),
)

核心操作​

Get 操作​

从缓存获取数据时，NodeShardedCache 会自动将请求路由到正确的节点：

func (nsc *NodeShardedCache) Get(ctx context.Context, key string) (any, error) {
    // 检查是否有可用节点
    if nsc.GetNodeCount() == 0 {
        return nil, ferr.ErrNoNodesAvailable
    }

    // 获取应该存储该键的所有节点（主节点和备份节点）
    nodeIDs := nsc.ring.GetN(key, nsc.replicaFactor)
    if len(nodeIDs) == 0 {
        return nil, ferr.ErrNoNodesAvailable
    }

    // 复制必要的节点信息以避免锁竞争
    nsc.mu.RLock()
    nodes := make([]cache.Cache, 0, len(nodeIDs))
    for _, id := range nodeIDs {
        if cache, exists := nsc.nodes[id]; exists {
            nodes = append(nodes, cache)
        }
    }
    nsc.mu.RUnlock()

    // 首先从主节点尝试获取
    if len(nodes) > 0 {
        if val, err := nodes[0].Get(ctx, key); err == nil {
            return val, nil
        }
    }

    // 如果主节点失败，尝试从备份节点获取
    for i := 1; i < len(nodes) && i < len(nodeIDs); i++ {
        if val, err := nodes[i].Get(ctx, key); err == nil {
            return val, nil
        }
    }

    return nil, ferr.ErrKeyNotFound
}

Set 操作​

设置数据时，NodeShardedCache 会将数据写入主节点和所有备份节点：

func (nsc *NodeShardedCache) Set(ctx context.Context, key string, val any, expiration time.Duration) error {
    // 获取应该存储该键的所有节点
    nodeIDs := nsc.ring.GetN(key, nsc.replicaFactor)
    if len(nodeIDs) == 0 {
        return ferr.ErrNoNodesAvailable
    }

    // 复制节点信息
    nsc.mu.RLock()
    nodes := make(map[string]cache.Cache, len(nodeIDs))
    for _, id := range nodeIDs {
        if cache, exists := nsc.nodes[id]; exists {
            nodes[id] = cache
        }
    }
    nsc.mu.RUnlock()

    // 追踪每个节点的错误
    errs := make([]error, 0)
    mainNodeID := nodeIDs[0]

    // 在主节点设置值
    if mainNode, ok := nodes[mainNodeID]; ok {
        if err := mainNode.Set(ctx, key, val, expiration); err != nil {
            errs = append(errs, fmt.Errorf("main node %s: %w", mainNodeID, err))
        }
    }

    // 在备份节点设置值
    for i := 1; i < len(nodeIDs) && i < nsc.replicaFactor; i++ {
        id := nodeIDs[i]
        if node, ok := nodes[id]; ok {
            if err := node.Set(ctx, key, val, expiration); err != nil {
                errs = append(errs, fmt.Errorf("backup node %s: %w", id, err))
            }
        }
    }

    // 只要主节点成功，就认为操作成功
    if len(errs) > 0 {
        if len(errs) == len(nodes) {
            return fmt.Errorf("all nodes failed: %v", errs)
        }
        // 检查主节点是否失败
        for _, err := range errs {
            if err.Error() == fmt.Sprintf("main node %s: %s", mainNodeID, ferr.ErrKeyNotFound.Error()) {
                return err
            }
        }
    }

    return nil
}

Del 操作​

删除数据时，NodeShardedCache 会在所有可能包含该键的节点上执行删除操作：

func (nsc *NodeShardedCache) Del(ctx context.Context, key string) error {
    // 获取可能包含该键的所有节点
    nodeIDs := nsc.ring.GetN(key, nsc.replicaFactor)
    if len(nodeIDs) == 0 {
        return ferr.ErrNoNodesAvailable
    }

    // 从所有节点删除数据
    // 实现与 Set 类似
}

节点管理​

添加节点​

可以在运行时向分片缓存添加新节点：

// 创建一个新的缓存节点
newNode := cache.NewMemoryCache()

// 添加到分片缓存，指定权重为2
err := shardedCache.AddNode("node4", newNode, 2)
if err != nil {
    // 处理错误
}

添加节点时会发生以下操作：

  1. 将新节点添加到一致性哈希环中
  2. 根据节点权重创建虚拟节点
  3. 环上的数据分布会自动调整，节点越多，每个节点迁移的数据量越少

移除节点​

同样可以在运行时移除节点：

// 从分片缓存移除节点
err := shardedCache.RemoveNode("node2")
if err != nil {
    // 处理错误
}

移除节点时：

  1. 节点从一致性哈希环上移除
  2. 原本分配给该节点的键将自动重新分配到环上的其他节点
  3. 如果没有副本机制，原节点的数据将不可访问；有副本时可从备份节点恢复

一致性哈希的应用​

NodeShardedCache 内部使用一致性哈希算法来分配数据：

// ConsistentHash 一致性哈希环实现
type ConsistentHash struct {
    hashFunc     HashFunc      // 哈希函数
    replicas     int           // 每个物理节点的虚拟节点数量
    sortedHashes []uint32      // 已排序的哈希值
    hashMap      map[uint32]string  // 哈希值到节点的映射
    weights      map[string]int     // 节点权重
    nodeHashes   map[string][]uint32 // 节点到其哈希值的映射
    mu           sync.RWMutex        // 保护并发访问
}

虚拟节点机制​

为了提高数据分布均匀性，每个物理节点会在哈希环上对应多个虚拟节点：

// 为节点创建虚拟节点
for i := 0; i < c.replicas*weight; i++ {
    virtualNodeName := fmt.Sprintf("%s:%d", node, i)
    hash := c.hashFunc([]byte(virtualNodeName))
    c.sortedHashes = append(c.sortedHashes, hash)
    c.hashMap[hash] = node
    nodeHashes = append(nodeHashes, hash)
}

默认情况下，FyerCache 使用 100 个虚拟节点，可以通过 WithNodeHashReplicas 选项调整。

节点权重​

不同节点可以设置不同的权重，代表其在环上占据的虚拟节点数量比例：

// 添加三个节点，权重不同
shardedCache.AddNode("powerful-node", node1, 3)  // 权重为3
shardedCache.AddNode("medium-node", node2, 2)    // 权重为2
shardedCache.AddNode("small-node", node3, 1)     // 权重为1

这种设计使得权重为 3 的节点会接收到约 3 倍于权重为 1 的节点的请求量，适用于配置不同的节点。

副本因子与容错机制​

副本因子决定每个数据项被存储在多少个节点上，默认为 1（不复制）：

// 创建副本因子为2的分片缓存
shardedCache := cache.NewNodeShardedCache(
    cache.WithNodeReplicaFactor(2),
)

容错机制​

当节点失败时，系统会尝试从备份节点获取数据：

  1. 首先尝试从主节点获取数据
  2. 如果主节点失败，依次尝试从备份节点获取
  3. 只有当所有节点都失败时才返回错误
// 如果主节点失败，尝试从备份节点获取
for i := 1; i < len(nodes) && i < len(nodeIDs); i++ {
    if val, err := nodes[i].Get(ctx, key); err == nil {
        return val, nil
    }
}

这种机制保证了即使某些节点失效，系统仍能提供服务。

使用场景示例​

基础示例​

package main

import (
    "context"
    "fmt"
    "time"

    "github.com/fyerfyer/fyer-cache/cache"
)

func main() {
    // 创建分片缓存
    shardedCache := cache.NewNodeShardedCache()
    
    // 添加三个缓存节点
    node1 := cache.NewMemoryCache()
    node2 := cache.NewMemoryCache()
    node3 := cache.NewMemoryCache()
    
    shardedCache.AddNode("node1", node1, 1)
    shardedCache.AddNode("node2", node2, 1)
    shardedCache.AddNode("node3", node3, 1)
    
    ctx := context.Background()
    
    // 存储数据 - 会自动路由到正确的节点
    for i := 1; i <= 10; i++ {
        key := fmt.Sprintf("key%d", i)
        err := shardedCache.Set(ctx, key, fmt.Sprintf("value%d", i), time.Hour)
        if err != nil {
            fmt.Printf("Failed to set %s: %v\n", key, err)
        }
    }
    
    // 读取数据 - 会自动从正确的节点获取
    for i := 1; i <= 10; i++ {
        key := fmt.Sprintf("key%d", i)
        val, err := shardedCache.Get(ctx, key)
        if err != nil {
            fmt.Printf("Failed to get %s: %v\n", key, err)
        } else {
            fmt.Printf("%s => %s\n", key, val)
        }
    }
}

带容错的示例​

// 创建副本因子为2的分片缓存
shardedCache := cache.NewNodeShardedCache(
    cache.WithNodeReplicaFactor(2),
    cache.WithNodeHashReplicas(200),
)

// 添加足够的节点以支持副本因子
shardedCache.AddNode("node1", node1, 1)
shardedCache.AddNode("node2", node2, 1)
shardedCache.AddNode("node3", node3, 1)

// 设置值将自动复制到多个节点
shardedCache.Set(ctx, "important-key", importantData, time.Hour)

// 即使主节点故障，也能获取数据
// 模拟节点1故障
shardedCache.RemoveNode("node1")

// 仍然可以获取数据，因为它存在于其他节点上
data, err := shardedCache.Get(ctx, "important-key")
