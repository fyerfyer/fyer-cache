Cache Consistency Strategy

FyerCache 提供多种缓存一致性策略，每种策略适用于不同的场景和要求。

核心接口设计​

ConsistencyStrategy 接口​

所有一致性策略都实现了 ConsistencyStrategy 接口：

// ConsistencyStrategy 定义缓存一致性策略接口
type ConsistencyStrategy interface {
    // Get 从缓存获取数据，缓存不存在时从数据源获取
    Get(ctx context.Context, key string) (any, error)

    // Set 设置缓存数据并根据策略同步到数据源
    Set(ctx context.Context, key string, value any, expiration time.Duration) error

    // Del 删除缓存和数据源中的数据
    Del(ctx context.Context, key string) error

    // Invalidate 使缓存失效但不影响数据源
    Invalidate(ctx context.Context, key string) error
}

DataSource 接口​

一致性策略需要与后端数据源交互，这由 DataSource 接口定义：

// DataSource 定义数据源接口
type DataSource interface {
    // Load 从数据源加载数据
    Load(ctx context.Context, key string) (any, error)

    // Store 将数据保存到数据源
    Store(ctx context.Context, key string, value any) error

    // Remove 从数据源删除数据
    Remove(ctx context.Context, key string) error
}

Cache-Aside 策略​

Cache-Aside（旁路缓存）是最常用的缓存策略之一，其工作原理是：

  • 读取时先查缓存，缓存未命中则从数据源加载并写入缓存
  • 写入时先更新数据源，再更新缓存
  • 删除时先删除缓存，再删除数据源

实现细节​

FyerCache 中的 Cache-Aside 策略实现如下：

// NewCacheAsideStrategy 创建Cache Aside策略实例
func NewCacheAsideStrategy(cache cache.Cache, dataSource DataSource, opts ...Option) *CacheAsideStrategy {
    options := defaultCacheAsideOptions()

    // 应用选项
    for _, opt := range opts {
        opt(options)
    }

    return &CacheAsideStrategy{
        cache:      cache,
        dataSource: dataSource,
        options:    options,
    }
}

Get 操作流程​

// Get 从缓存获取数据，缓存未命中时从数据源获取
func (c *CacheAsideStrategy) Get(ctx context.Context, key string) (any, error) {
    // 首先尝试从缓存获取
    value, err := c.cache.Get(ctx, key)
    if err == nil {
        // 缓存命中，直接返回
        return value, nil
    }

    // 如果错误不是"键不存在"，则返回错误
    if err != ferr.ErrKeyNotFound {
        return nil, err
    }

    // 从数据源获取
    value, err = c.dataSource.Load(ctx, key)
    if err != nil {
        return nil, fmt.Errorf("failed to load from data source: %w", err)
    }

    // 如果配置了缓存未命中时写入，将数据写入缓存
    if c.options.WriteOnMiss {
        // 使用默认过期时间写入缓存
        // 这里不处理写入缓存的错误，即使写入失败也返回查询结果
        _ = c.cache.Set(ctx, key, value, c.options.DefaultTTL)
    }

    return value, nil
}

Set 操作流程​

// Set 设置缓存数据并同步到数据源
// 在Cache Aside模式中，先更新数据源再更新缓存
func (c *CacheAsideStrategy) Set(ctx context.Context, key string, value any, expiration time.Duration) error {
    // 先更新数据源
    if err := c.dataSource.Store(ctx, key, value); err != nil {
        return fmt.Errorf("failed to store in data source: %w", err)
    }

    // 再更新缓存
    if err := c.cache.Set(ctx, key, value, expiration); err != nil {
        // 即使缓存更新失败，数据源已更新成功，我们仍认为操作成功
        // 但记录错误
        return fmt.Errorf("data source updated but cache set failed: %w", err)
    }

    return nil
}

Del 操作流程​

// Del 删除缓存和数据源中的数据
// 在Cache Aside模式中，先删除缓存再删除数据源
func (c *CacheAsideStrategy) Del(ctx context.Context, key string) error {
    // 先删除缓存
    _ = c.cache.Del(ctx, key)

    // 再删除数据源
    if err := c.dataSource.Remove(ctx, key); err != nil {
        return fmt.Errorf("failed to remove from data source: %w", err)
    }

    return nil
}

使用示例​

package main

import (
    "context"
    "fmt"
    "time"
    
    "github.com/fyerfyer/fyer-cache/cache"
    "github.com/fyerfyer/fyer-cache/cache/consistency"
)

// 简单的数据源实现
type UserDB struct {
    users map[string]string
}

func (db *UserDB) Load(ctx context.Context, key string) (any, error) {
    if val, exists := db.users[key]; exists {
        return val, nil
    }
    return nil, fmt.Errorf("user not found")
}

func (db *UserDB) Store(ctx context.Context, key string, value any) error {
    if str, ok := value.(string); ok {
        db.users[key] = str
        return nil
    }
    return fmt.Errorf("invalid value type")
}

func (db *UserDB) Remove(ctx context.Context, key string) error {
    delete(db.users, key)
    return nil
}

func main() {
    // 初始化缓存和数据源
    memCache := cache.NewMemoryCache()
    userDB := &UserDB{users: map[string]string{
        "user:1001": "John Doe",
        "user:1002": "Jane Smith",
    }}
    
    // 创建 Cache-Aside 策略
    cacheStrategy := consistency.NewCacheAsideStrategy(
        memCache, 
        userDB,
        consistency.WithCacheAsideTTL(10*time.Minute),
    )
    
    ctx := context.Background()
    
    // 使用策略获取数据，如果缓存未命中，会从数据源加载
    value, err := cacheStrategy.Get(ctx, "user:1001")
    if err != nil {
        fmt.Printf("Error: %v\n", err)
        return
    }
    fmt.Printf("User: %v\n", value)
    
    // 写入数据，会同时更新数据源和缓存
    err = cacheStrategy.Set(ctx, "user:1003", "Robert Brown", 10*time.Minute)
    if err != nil {
        fmt.Printf("Error: %v\n", err)
        return
    }
    
    // 删除数据，会同时从数据源和缓存中删除
    err = cacheStrategy.Del(ctx, "user:1002")
    if err != nil {
        fmt.Printf("Error: %v\n", err)
        return
    }
}

配置选项​

Cache-Aside 策略支持以下配置选项：

// WithCacheAsideTTL 设置 Cache Aside 策略的默认TTL
func WithCacheAsideTTL(ttl time.Duration) Option {
    return func(o interface{}) {
        if ca, ok := o.(*CacheAsideOptions); ok {
            ca.DefaultTTL = ttl
        }
    }
}

// WithCacheAsideWriteOnMiss 设置是否在未命中时写入缓存
func WithCacheAsideWriteOnMiss(writeOnMiss bool) Option {
    return func(o interface{}) {
        if ca, ok := o.(*CacheAsideOptions); ok {
            ca.WriteOnMiss = writeOnMiss
        }
    }
}

Write-Through 策略​

Write-Through（直写）策略确保数据在写入缓存的同时也写入数据源，从而保证数据的一致性：

  • 读取时先查缓存，缓存未命中则从数据源加载并写入缓存
  • 写入时先更新数据源，再更新缓存
  • 删除时先删除数据源，再删除缓存

实现细节​

// NewWriteThroughStrategy 创建 Write Through 策略实例
func NewWriteThroughStrategy(cache cache.Cache, dataSource DataSource, opts ...Option) *WriteThroughStrategy {
    options := defaultWriteThroughOptions()

    // 应用选项
    for _, opt := range opts {
        opt(options)
    }

    return &WriteThroughStrategy{
        cache:      cache,
        dataSource: dataSource,
        options:    options,
    }
}

Set 操作与延迟写入​

Write-Through 策略的特色是支持延迟写入，可以提高写入性能：

// Set 设置缓存数据并同步到数据源
// 在Write Through模式中，同时更新数据源和缓存
func (w *WriteThroughStrategy) Set(ctx context.Context, key string, value any, expiration time.Duration) error {
    // 如果启用了延迟写入
    if w.options.DelayedWrite {
        // 先更新缓存
        if err := w.cache.Set(ctx, key, value, expiration); err != nil {
            return fmt.Errorf("failed to set cache: %w", err)
        }

        // 异步更新数据源
        go func() {
            // 创建新的上下文，避免使用可能已取消的上下文
            newCtx := context.Background()

            // 延迟一段时间再写入数据源
            time.Sleep(w.options.WriteDelay)

            // 更新数据源
            if err := w.dataSource.Store(newCtx, key, value); err != nil {
                // 实际应用中应当有错误处理日志
            }
        }()

        return nil
    }

    // 标准Write Through：同时更新数据源和缓存
    // 先更新数据源
    if err := w.dataSource.Store(ctx, key, value); err != nil {
        return fmt.Errorf("failed to store in data source: %w", err)
    }

    // 再更新缓存
    if err := w.cache.Set(ctx, key, value, expiration); err != nil {
        // 数据源已更新，但缓存更新失败
        // 此时需要移除缓存项以避免不一致
        _ = w.cache.Del(ctx, key)
        return fmt.Errorf("data source updated but cache set failed: %w", err)
    }

    return nil
}

使用示例​

// 创建 Write-Through 策略
writeThrough := consistency.NewWriteThroughStrategy(
    memCache,
    userDB,
    consistency.WithWriteThroughTTL(10*time.Minute),
    consistency.WithDelayedWrite(true, 100*time.Millisecond), // 使用延迟写入
)

// 使用策略写入，会先写缓存，延迟写入数据源
err := writeThrough.Set(ctx, "user:1004", "Alice Green", 10*time.Minute)

配置选项​

Write-Through 策略支持以下配置选项：

// WithWriteThroughTTL 设置 Write Through 策略的默认TTL
func WithWriteThroughTTL(ttl time.Duration) Option {
    return func(o interface{}) {
        if wt, ok := o.(*WriteThroughOptions); ok {
            wt.DefaultTTL = ttl
        }
    }
}

// WithDelayedWrite 设置是否使用延迟写入及延迟时间
func WithDelayedWrite(delayed bool, delay time.Duration) Option {
    return func(o interface{}) {
        if wt, ok := o.(*WriteThroughOptions); ok {
            wt.DelayedWrite = delayed
            if delay > 0 {
                wt.WriteDelay = delay
            }
        }
    }
}

基于消息队列的通知策略​

MQNotifierStrategy（消息队列通知策略）利用消息队列在分布式环境中传播缓存变更，确保多个缓存节点的一致性：

  • 缓存更改时发送消息通知其他节点
  • 接收节点根据消息类型采取相应操作（更新或失效）
  • 支持消息重传和失败恢复

实现细节​

// NewMQNotifierStrategy 创建基于消息队列通知的一致性策略
func NewMQNotifierStrategy(cache cache.Cache, dataSource DataSource, publisher MessagePublisher, opts ...Option) *MQNotifierStrategy {
    options := defaultMQNotifierOptions()

    // 应用选项
    for _, opt := range opts {
        opt(options)
    }

    strategy := &MQNotifierStrategy{
        cache:      cache,
        dataSource: dataSource,
        publisher:  publisher,
        options:    options,
    }

    // 自动订阅消息处理
    err := publisher.Subscribe(options.Topic, strategy.handleMessage)
    if err != nil {
        // 实际应用应当有日志记录
    }

    return strategy
}

发布缓存更新消息​

// publishWithRetry 尝试发布消息，支持重试
func (m *MQNotifierStrategy) publishWithRetry(ctx context.Context, message CacheMessage) error {
    // 序列化消息
    msgBytes, err := json.Marshal(message)
    if err != nil {
        return fmt.Errorf("failed to marshal cache message: %w", err)
    }

    // 第一次尝试
    err = m.publisher.Publish(ctx, m.options.Topic, msgBytes)
    if err == nil {
        return nil
    }

    // 失败后进行重试
    retries := 0
    for retries < m.options.MaxRetries {
        // 等待重试间隔
        time.Sleep(m.options.RetryDelay)
        
        // 再次尝试
        err = m.publisher.Publish(ctx, m.options.Topic, msgBytes)
        if err == nil {
            return nil
        }
        
        retries++
    }
    
    return fmt.Errorf("failed to publish message after %d retries: %w", m.options.MaxRetries, err)
}

处理接收到的消息​

// handleMessage 处理接收到的缓存消息
func (m *MQNotifierStrategy) handleMessage(msg []byte) {
    var message CacheMessage
    err := json.Unmarshal(msg, &message)
    if err != nil {
        // 日志记录解析错误
        log.Printf("Failed to unmarshal cache message: %v", err)
        return
    }

    // 创建带超时的上下文
    ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
    defer cancel()

    switch message.Type {
    case MsgTypeInvalidate:
        // 处理缓存失效消息
        _ = m.cache.Del(ctx, message.Key)
        
    case MsgTypeUpdate:
        // 处理缓存更新消息
        if len(message.Data) > 0 {
            var value interface{}
            // 尝试反序列化值
            if err := json.Unmarshal(message.Data, &value); err == nil {
                _ = m.cache.Set(ctx, message.Key, value, m.options.DefaultTTL)
            }
        }
    }
}

使用示例​

// 简单实现一个消息发布者
type SimplePublisher struct {
    subscribers []func([]byte)
}

func (p *SimplePublisher) Publish(ctx context.Context, topic string, msg []byte) error {
    for _, sub := range p.subscribers {
        sub(msg)
    }
    return nil
}

func (p *SimplePublisher) Subscribe(topic string, handler func(msg []byte)) error {
    p.subscribers = append(p.subscribers, handler)
    return nil
}

func (p *SimplePublisher) Close() error {
    return nil
}

// 创建发布者
publisher := &SimplePublisher{}

// 创建基于消息队列的通知策略
mqStrategy := consistency.NewMQNotifierStrategy(
    memCache,
    userDB,
    publisher,
    consistency.WithMQTopic("cache:invalidate"),
    consistency.WithMQNotifierTTL(10*time.Minute),
)

// 使用策略，写入或删除操作会通过消息队列通知其他节点
err := mqStrategy.Set(ctx, "user:1005", "Tom Wilson", 10*time.Minute)

配置选项​

MQ通知策略支持以下配置选项：

// WithMQTopic 设置消息主题
func WithMQTopic(topic string) Option {
    return func(o interface{}) {
        if mq, ok := o.(*MQNotifierOptions); ok {
            mq.Topic = topic
        }
    }
}

// WithMQRetry 设置重试策略
func WithMQRetry(maxRetries int, retryDelay time.Duration) Option {
    return func(o interface{}) {
        if mq, ok := o.(*MQNotifierOptions); ok {
            if maxRetries >= 0 {
                mq.MaxRetries = maxRetries
            }
            if retryDelay > 0 {
                mq.RetryDelay = retryDelay
            }
        }
    }
}

// WithMQNotifierTTL 设置 MQ 通知器的默认TTL
func WithMQNotifierTTL(ttl time.Duration) Option {
    return func(o interface{}) {
        if mq, ok := o.(*MQNotifierOptions); ok {
            mq.DefaultTTL = ttl
        }
    }
}
