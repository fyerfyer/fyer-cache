Cluster-Management API

基础集群节点​

Node​

Node 是 FyerCache 集群的基础构建块，封装了单个缓存节点的功能，包括成员管理、通信和状态管理。

创建节点​

func NewNode(id string, cache cache.Cache, options ...NodeOption) (*Node, error)

参数：

  • id - 节点唯一标识符
  • cache - 本地缓存实例
  • options - 节点配置选项

返回值：

  • 创建的节点实例和可能的错误

示例：

// 创建内存缓存实例
localCache := cache.NewMemoryCache()

// 创建集群节点
node, err := cluster.NewNode(
    "node-1",
    localCache,
    cluster.WithBindAddr(":7946"),
    cluster.WithGossipInterval(500 * time.Millisecond),
    cluster.WithProbeInterval(1 * time.Second),
)

主要方法​

// 启动节点服务
func (n *Node) Start() error

// 停止节点服务
func (n *Node) Stop() error

// 加入集群
func (n *Node) Join(seedNodeAddr string) error

// 离开集群
func (n *Node) Leave() error

// 获取集群成员列表
func (n *Node) Members() []cache.NodeInfo

// 判断当前节点是否为协调者

// 获取事件通知通道
func (n *Node) Events() <-chan cache.ClusterEvent

// 获取节点的成员管理器
func (n *Node) Membership() *Membership

事件使用示例​

// 订阅集群事件
eventChannel := node.Events()
go func() {
    for event := range eventChannel {
        switch event.Type {
        case cache.EventNodeJoin:
            fmt.Printf("Node joined: %s\n", event.NodeID)
        case cache.EventNodeLeave:
            fmt.Printf("Node left: %s\n", event.NodeID)
        case cache.EventNodeFailed:
            fmt.Printf("Node failed: %s, reason: %v\n", event.NodeID, event.Details)
        case cache.EventNodeRecovered:
            fmt.Printf("Node recovered: %s\n", event.NodeID)
        case cache.EventSyncCompleted:
            fmt.Printf("Sync completed with node: %s\n", event.NodeID)
        }
    }
}()

完整示例：主从架构集群​

package main

import (
	"context"
	"fmt"
	"log"
	"time"

	"github.com/fyerfyer/fyer-cache/cache"
	"github.com/fyerfyer/fyer-cache/cache/cluster"
	"github.com/fyerfyer/fyer-cache/cache/cluster/integration"
	"github.com/fyerfyer/fyer-cache/cache/replication"
)

func main() {
	// 创建三个节点的主从集群
	node1, err := createLeaderFollowerCluster("node1", ":7946", ":8080", replication.RoleLeader)
	if err != nil {
		log.Fatalf("Failed to create node1: %v", err)
	}
	
	node2, err := createLeaderFollowerCluster("node2", ":7947", ":8081", replication.RoleFollower)
	if err != nil {
		log.Fatalf("Failed to create node2: %v", err)
	}
	
	node3, err := createLeaderFollowerCluster("node3", ":7948", ":8082", replication.RoleFollower)
	if err != nil {
		log.Fatalf("Failed to create node3: %v", err)
	}
	
	// 启动所有节点
	if err := node1.Start(); err != nil {
		log.Fatalf("Failed to start node1: %v", err)
	}
	if err := node2.Start(); err != nil {
		log.Fatalf("Failed to start node2: %v", err)
	}
	if err := node3.Start(); err != nil {
		log.Fatalf("Failed to start node3: %v", err)
	}
	
	// 确保优雅退出
	defer func() {
		node1.Stop()
		node2.Stop()
		node3.Stop()
	}()
	
	// 形成集群
	node1.Join("localhost:7946")  // 第一个节点自引导
	time.Sleep(500 * time.Millisecond)
	
	node2.Join("localhost:7946")  // 其他节点加入第一个节点
	time.Sleep(500 * time.Millisecond)
	
	node3.Join("localhost:7946")
	
	// 等待集群稳定
	time.Sleep(2 * time.Second)
	
	// 为从节点设置领导者
	node2.SetLeader("node1", "localhost:8080")
	node3.SetLeader("node1", "localhost:8080")
	
	// 写入数据到领导者
	ctx := context.Background()
	err = node1.Set(ctx, "key1", "value1", 10*time.Minute)
	if err != nil {
		log.Printf("Error writing to leader: %v", err)
	} else {
		fmt.Println("Data written to leader node")
	}
	
	// 等待复制完成
	time.Sleep(1 * time.Second)
	
	// 从跟随者节点读取数据
	value, err := node2.Get(ctx, "key1")
	if err != nil {
		log.Printf("Error reading from node2: %v", err)
	} else {
		fmt.Printf("Value from node2: %v\n", value)
	}
	
	value, err = node3.Get(ctx, "key1")
	if err != nil {
		log.Printf("Error reading from node3: %v", err)
	} else {
		fmt.Printf("Value from node3: %v\n", value)
	}
}

// 创建主从集群节点
func createLeaderFollowerCluster(nodeID, clusterAddr, syncerAddr string, role replication.ReplicationRole) (*integration.LeaderFollowerCluster, error) {
	localCache := cache.NewMemoryCache()
	
	return integration.NewLeaderFollowerCluster(
		nodeID, 
		clusterAddr, 
		localCache,
		[]cluster.NodeOption{
			cluster.WithNodeID(nodeID),
			cluster.WithGossipInterval(500 * time.Millisecond),
			cluster.WithProbeInterval(1 * time.Second),
		},
		[]integration.LeaderFollowerClusterOption{
			integration.WithSyncerAddress(syncerAddr),
			integration.WithInitialRole(role),
			integration.WithHeartbeatInterval(500 * time.Millisecond),
		},
	)
}

完整示例：分片集群​

package main

import (
    "context"
    "fmt"
    "log"
    "time"

    "github.com/fyerfyer/fyer-cache/cache"
    "github.com/fyerfyer/fyer-cache/cache/cluster"
    "github.com/fyerfyer/fyer-cache/cache/cluster/integration"
)

func main() {
    // 创建三个节点的分片集群
    node1, err := createShardedClusterCache("node1", ":7946")
    if err != nil {
        log.Fatalf("Failed to create node1: %v", err)
    }
    
    node2, err := createShardedClusterCache("node2", ":7947")
    if err != nil {
        log.Fatalf("Failed to create node2: %v", err)
    }
    
    node3, err := createShardedClusterCache("node3", ":7948")
    if err != nil {
        log.Fatalf("Failed to create node3: %v", err)
    }
    
    // 启动所有节点
    if err := node1.Start(); err != nil {
        log.Fatalf("Failed to start node1: %v", err)
    }
    if err := node2.Start(); err != nil {
        log.Fatalf("Failed to start node2: %v", err)
    }
    if err := node3.Start(); err != nil {
        log.Fatalf("Failed to start node3: %v", err)
    }
    
    // 确保优雅退出
    defer func() {
        node1.Stop()
        node2.Stop()
        node3.Stop()
    }()
    
    // 形成集群
    node1.Join("localhost:7946")  // 第一个节点自引导
    time.Sleep(500 * time.Millisecond)
    
    node2.Join("localhost:7946")  // 其他节点加入第一个节点
    time.Sleep(500 * time.Millisecond)
    
    node3.Join("localhost:7946")
    
    // 等待集群稳定
    time.Sleep(2 * time.Second)
    
    // 存储一些数据
    ctx := context.Background()
    for i := 1; i <= 10; i++ {
        key := fmt.Sprintf("key%d", i)
        value := fmt.Sprintf("value%d", i)
        
        // 随机选择一个节点进行写入
        // 在实际的分片集群中，数据会根据一致性哈希自动路由到正确的节点
        var node *integration.ShardedClusterCache
        switch i % 3 {
        case 0:
            node = node1
        case 1:
            node = node2
        case 2:
            node = node3
        }
        
        err := node.Set(ctx, key, value, time.Hour)
        if err != nil {
            log.Printf("Failed to set %s: %v", key, err)
        } else {
            fmt.Printf("Set %s to %s\n", key, value)
        }
    }
    
    // 从不同节点读取数据
    time.Sleep(500 * time.Millisecond)
    
    for i := 1; i <= 10; i++ {
        key := fmt.Sprintf("key%d", i)
        
        // 从随机节点读取数据
        var node *integration.ShardedClusterCache
        switch i % 3 {
        case 0:
            node = node3  // 故意使用与写入不同的节点
        case 1:
            node = node1
        case 2:
            node = node2
        }
        
        value, err := node.Get(ctx, key)
        if err != nil {
            log.Printf("Failed to get %s: %v", key, err)
        } else {
            fmt.Printf("Got %s = %v\n", key, value)
        }
    }
}

// 创建分片集群节点
func createShardedClusterCache(nodeID, addr string) (*integration.ShardedClusterCache, error) {
    localCache := cache.NewMemoryCache()
    
    return integration.NewShardedClusterCache(
        localCache,
        nodeID,
        addr,
        []cluster.NodeOption{
            cluster.WithNodeID(nodeID),
            cluster.WithGossipInterval(500 * time.Millisecond),
            cluster.WithProbeInterval(1 * time.Second),
        },
        []integration.ShardedClusterCacheOption{
            integration.WithVirtualNodeCount(100),
            integration.WithReplicaFactor(2),
        },
    )
}
