安装​

使用 Go 的包管理工具安装 FyerCache：

go get -u github.com/fyerfyer/fyer-cache

确保您的项目使用 Go Modules 进行依赖管理，并且 Go 版本 >= 1.19。

示例 1：简单内存缓存​

最简单的用法是使用内存缓存，适合单机应用或作为本地缓存层。

package main

import (
    "context"
    "fmt"
    "time"

    "github.com/fyerfyer/fyer-cache/cache"
)

func main() {
    // 创建一个简单的内存缓存
    memCache := cache.NewMemoryCache()
    ctx := context.Background()

    // 设置缓存，5分钟过期
    err := memCache.Set(ctx, "user:1001", "John Doe", 5*time.Minute)
    if err != nil {
        fmt.Printf("Failed to set cache: %v\n", err)
        return
    }
    
    // 获取缓存值
    val, err := memCache.Get(ctx, "user:1001")
    if err != nil {
        fmt.Printf("Failed to get from cache: %v\n", err)
        return
    }
    fmt.Printf("User: %v\n", val)
    
    // 删除缓存
    err = memCache.Del(ctx, "user:1001")
    if err != nil {
        fmt.Printf("Failed to delete from cache: %v\n", err)
        return
    }
    fmt.Println("Cache entry deleted")
    
    // 尝试获取已删除的缓存
    _, err = memCache.Get(ctx, "user:1001")
    if err != nil {
        fmt.Printf("Expected error: %v\n", err)
    }
}

示例 2：配置内存缓存​

内存缓存提供多种配置选项，让您可以根据需求自定义缓存行为：

package main

import (
    "context"
    "fmt"
    "runtime"
    "time"

    "github.com/fyerfyer/fyer-cache/cache"
)

func main() {
    // 使用多个选项配置内存缓存
    memCache := cache.NewMemoryCache(
        // 设置分片数量以减少锁竞争（建议设置为CPU核心数的倍数）
        cache.WithShardCount(32),
        
        // 设置淘汰回调，在条目被淘汰时执行
        cache.WithEvictionCallback(func(key string, value any) {
            fmt.Printf("Evicted: %s\n", key)
        }),
        
        // 设置清理间隔，定期清理过期项
        cache.WithCleanupInterval(5*time.Minute),
        
        // 启用异步清理，减少清理对主线程性能的影响
        cache.WithAsyncCleanup(true),
        
        // 设置异步清理的工作协程数
        cache.CacheWithWorkerCount(runtime.NumCPU()),
    )
    
    ctx := context.Background()
    
    // 添加一个短期缓存项
    memCache.Set(ctx, "temp", "temporary value", 2*time.Second)
    
    // 等待项目过期并被清理
    time.Sleep(3 * time.Second)
    
    // 检查项目是否已被清理
    _, err := memCache.Get(ctx, "temp")
    if err != nil {
        fmt.Println("Item expired and cleaned up as expected")
    }
    
    // 请务必在应用结束前关闭缓存以释放资源
    memCache.Close()
}

示例 3：分片缓存​

分片缓存可以显著提高并发性能，特别是在多核系统上：

package main

import (
    "context"
    "fmt"
    "sync"
    "time"

    "github.com/fyerfyer/fyer-cache/cache"
)

func main() {
    // 创建32个分片的内存缓存
    shardedCache := cache.NewMemoryCache(cache.WithShardCount(32))
    ctx := context.Background()

    // 批量写入示例
    var wg sync.WaitGroup
    for i := 0; i < 1000; i++ {
        wg.Add(1)
        go func(id int) {
            defer wg.Done()
            key := fmt.Sprintf("key:%d", id)
            value := fmt.Sprintf("value:%d", id)
            err := shardedCache.Set(ctx, key, value, 5*time.Minute)
            if err != nil {
                fmt.Printf("Error setting %s: %v\n", key, err)
            }
        }(i)
    }
    wg.Wait()
    
    fmt.Println("Batch write completed")
    
    // 批量读取示例
    hitCount := 0
    var mu sync.Mutex
    
    for i := 0; i < 1000; i++ {
        wg.Add(1)
        go func(id int) {
            defer wg.Done()
            key := fmt.Sprintf("key:%d", id)
            if _, err := shardedCache.Get(ctx, key); err == nil {
                mu.Lock()
                hitCount++
                mu.Unlock()
            }
        }(i)
    }
    wg.Wait()
    
    fmt.Printf("Cache hit rate: %d%%\n", hitCount/10)
}

示例 4：基于节点的分片缓存​

节点分片缓存允许您在多个后端存储节点之间分发数据，这是构建分布式缓存的基础：

package main

import (
    "context"
    "fmt"

    "github.com/fyerfyer/fyer-cache/cache"
)

func main() {
    // 创建一个基于节点的分片缓存
    // 这里我们定义三个后端存储节点
    shardedCache := cache.NewNodeShardedCache()
    
    // 为每个节点创建独立的内存缓存
    node1 := cache.NewMemoryCache()
    node2 := cache.NewMemoryCache()
    node3 := cache.NewMemoryCache()
    
    // 添加节点到分片缓存，参数为：节点ID, 缓存实例, 权重
    err := shardedCache.AddNode("node1", node1, 1)
    if err != nil {
        panic(err)
    }
    
    err = shardedCache.AddNode("node2", node2, 1)
    if err != nil {
        panic(err)
    }
    
    err = shardedCache.AddNode("node3", node3, 2) // node3 的权重为2，将接收更多数据
    if err != nil {
        panic(err)
    }
    
    fmt.Printf("Created node sharded cache with %d nodes\n", shardedCache.GetNodeCount())
    
    ctx := context.Background()
    
    // 数据会根据键自动路由到正确的节点
    for i := 1; i <= 10; i++ {
        key := fmt.Sprintf("key%d", i)
        value := fmt.Sprintf("value%d", i)
        
        err := shardedCache.Set(ctx, key, value, 0) // 0 表示不过期
        if err != nil {
            fmt.Printf("Failed to set %s: %v\n", key, err)
        }
    }
    
    // 获取数据时，分片缓存会自动从正确的节点获取
    for i := 1; i <= 10; i++ {
        key := fmt.Sprintf("key%d", i)
        val, err := shardedCache.Get(ctx, key)
        if err != nil {
            fmt.Printf("Failed to get %s: %v\n", key, err)
        } else {
            fmt.Printf("%s => %s\n", key, val)
        }
    }
}

示例 5：分布式集群缓存​

这个示例展示如何创建一个简单的分片集群缓存，在多个节点之间分发数据：

package main

import (
    "context"
    "fmt"
    "time"

    "github.com/fyerfyer/fyer-cache/cache"
    "github.com/fyerfyer/fyer-cache/cache/cluster"
    "github.com/fyerfyer/fyer-cache/cache/cluster/integration"
)

func main() {
    // 创建两个节点的分片集群
    
    // 为每个节点创建本地缓存
    localCache1 := cache.NewMemoryCache()
    localCache2 := cache.NewMemoryCache()
    
    // 创建第一个节点
    node1, err := integration.NewShardedClusterCache(
        localCache1,
        "node1",
        "localhost:7001",
        []cluster.NodeOption{
            cluster.WithGossipInterval(200 * time.Millisecond),
        },
        []integration.ShardedClusterCacheOption{
            integration.WithVirtualNodeCount(100),
        },
    )
    if err != nil {
        panic(err)
    }
    
    // 创建第二个节点
    node2, err := integration.NewShardedClusterCache(
        localCache2,
        "node2",
        "localhost:7002",
        []cluster.NodeOption{
            cluster.WithGossipInterval(200 * time.Millisecond),
        },
        []integration.ShardedClusterCacheOption{
            integration.WithVirtualNodeCount(100),
        },
    )
    if err != nil {
        panic(err)
    }
    
    // 启动节点
    if err := node1.Start(); err != nil {
        panic(err)
    }
    if err := node2.Start(); err != nil {
        panic(err)
    }
    
    // 确保优雅退出
    defer func() {
        node1.Stop()
        node2.Stop()
    }()
    
    // 节点1形成自己的集群
    if err := node1.Join(node1.GetAddress()); err != nil {
        panic(err)
    }
    fmt.Printf("Node %s formed its own cluster\n", node1.GetNodeID())
    
    // 节点2加入节点1的集群
    if err := node2.Join(node1.GetAddress()); err != nil {
        panic(err)
    }
    fmt.Printf("Node %s joined the cluster\n", node2.GetNodeID())
    
    // 等待集群稳定
    fmt.Println("Waiting for cluster to stabilize...")
    time.Sleep(2 * time.Second)
    
    // 使用集群存储和检索数据
    ctx := context.Background()
    
    // 向集群写入数据
    fmt.Println("Storing data in the cluster...")
    err = node1.Set(ctx, "cluster-key", "cluster-value", 10*time.Minute)
    if err != nil {
        fmt.Printf("Failed to set data: %v\n", err)
    } else {
        fmt.Println("Data stored successfully")
    }
    
    // 从另一个节点读取数据
    fmt.Println("Retrieving data from the cluster...")
    val, err := node2.Get(ctx, "cluster-key")
    if err != nil {
        fmt.Printf("Failed to get data: %v\n", err)
    } else {
        fmt.Printf("Retrieved value: %v\n", val)
    }
}

示例 6：添加监控​

FyerCache 提供内置监控功能，可以轻松追踪缓存性能和使用情况：

package main

import (
    "context"
    "fmt"
    "time"

    "github.com/fyerfyer/fyer-cache/cache"
    "github.com/fyerfyer/fyer-cache/cache/metrics"
)

func main() {
    // 创建基础内存缓存
    memCache := cache.NewMemoryCache()
    
    // 使用监控装饰器增强缓存
    monitoredCache := metrics.NewMonitoredCache(
        memCache,
        metrics.WithMetricsServer(":8080"),      // 在8080端口暴露指标
        metrics.WithNamespace("myapp"),          // 指标命名空间
        metrics.WithSubsystem("usercache"),      // 指标子系统
        metrics.WithCollectInterval(10*time.Second), // 每10秒收集一次指标
    )
    
    // 启动监控服务器
    monitoredCache.Start()
    defer monitoredCache.Stop()
    
    fmt.Println("Metrics server started on :8080")
    fmt.Println("Visit http://localhost:8080/metrics to view Prometheus metrics")
    
    ctx := context.Background()
    
    // 执行一些缓存操作来生成指标
    for i := 0; i < 100; i++ {
        key := fmt.Sprintf("user:%d", i)
        monitoredCache.Set(ctx, key, fmt.Sprintf("User %d", i), 10*time.Minute)
    }
    
    // 读取一些缓存项，产生命中和未命中
    for i := 0; i < 150; i++ {
        key := fmt.Sprintf("user:%d", i)
        monitoredCache.Get(ctx, key)
    }
    
    // 显示命中率
    fmt.Printf("Current hit rate: %.2f%%\n", monitoredCache.HitRate()*100)
    
    fmt.Println("Press Ctrl+C to exit")
    
    // 使程序保持运行以便查看指标
    select {}
}

示例 7：API 服务器​

FyerCache 还提供了 HTTP API 服务器，方便通过 REST API 管理缓存：

package main

import (
    "fmt"
    "time"

    "github.com/fyerfyer/fyer-cache/cache"
    "github.com/fyerfyer/fyer-cache/cache/api"
)

func main() {
    // 创建内存缓存
    memCache := cache.NewMemoryCache()
    
    // 创建 API 服务器
    apiServer := api.NewAPIServer(
        memCache,
        api.WithBindAddress(":8081"),
        api.WithBasePath("/api/cache"),
        api.WithTimeouts(5*time.Second, 5*time.Second, 30*time.Second),
    )
    
    // 启动 API 服务器
    err := apiServer.Start()
    if err != nil {
        panic(err)
    }
    
    fmt.Println("API server started on :8081")
    fmt.Println("Available endpoints:")
    fmt.Println("  GET  /api/cache/stats - Get cache statistics")
    fmt.Println("  POST /api/cache/invalidate?key=X - Invalidate cache entry")
    
    // 使程序保持运行
    select {}
}
