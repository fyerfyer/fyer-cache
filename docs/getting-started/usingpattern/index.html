Using Pattern

本文档介绍 FyerCache 的核心功能和基本使用模式。

核心缓存接口​

所有 FyerCache 缓存实现都基于相同的核心接口：

// Cache 基本缓存接口
type Cache interface {
    Get(ctx context.Context, key string) (any, error)
    Set(ctx context.Context, key string, val any, expiration time.Duration) error
    Del(ctx context.Context, key string) error
}

这个简洁的接口使所有缓存实现保持一致的使用模式，无论是内存缓存还是分布式缓存。

使用内存缓存​

创建基础内存缓存​

package main

import (
    "context"
    "fmt"
    "time"

    "github.com/fyerfyer/fyer-cache/cache"
)

func main() {
    // 创建一个内存缓存，使用默认配置
    memCache := cache.NewMemoryCache()
    ctx := context.Background()

    // 设置缓存项，5分钟后过期
    err := memCache.Set(ctx, "user:1001", "John Doe", 5*time.Minute)
    if err != nil {
        fmt.Printf("Failed to set cache: %v\n", err)
        return
    }

    // 获取缓存项
    val, err := memCache.Get(ctx, "user:1001")
    if err != nil {
        fmt.Printf("Failed to get from cache: %v\n", err)
        return
    }
    fmt.Printf("User: %v\n", val)

    // 删除缓存项
    err = memCache.Del(ctx, "user:1001")
    if err != nil {
        fmt.Printf("Failed to delete from cache: %v\n", err)
        return
    }
}

配置内存缓存​

内存缓存支持多种配置选项，通过选项模式设置：

// 创建带有自定义配置的内存缓存
cache := cache.NewMemoryCache(
    // 设置分片数量，减少锁竞争
    cache.WithShardCount(32), 
    
    // 设置淘汰回调函数
    cache.WithEvictionCallback(func(key string, value any) {
        fmt.Printf("Evicted: %s\n", key)
    }),
    
    // 设置清理间隔
    cache.WithCleanupInterval(10*time.Minute),
    
    // 启用异步清理
    cache.WithAsyncCleanup(true),
    
    // 设置清理工作协程数量
    cache.CacheWithWorkerCount(runtime.NumCPU()),
)

异步清理过期项目​

当缓存中有大量项目时，同步清理可能会导致性能问题。FyerCache 提供异步清理功能：

// 使用异步清理创建缓存
cache := cache.NewMemoryCache(
    cache.WithAsyncCleanup(true),       // 启用异步清理
    cache.CacheWithWorkerCount(4),      // 4个清理工作协程
    cache.CacheWithQueueSize(100),      // 任务队列大小
    cache.WithCleanupInterval(5*time.Minute), // 每5分钟执行清理
)

// 使用完后记得关闭资源
defer cache.Close()

内存限制和 LRU 淘汰​

对于需要控制内存使用的场景，可以使用 MemCacheMemory：

// 创建一个基础内存缓存
baseCache := cache.NewMemoryCache()

// 创建一个有内存限制的缓存装饰器，限制为 100MB
memLimitedCache := cache.NewMemCacheMemory(baseCache, 100*1024*1024)

// 当内存使用超过限制时，会使用 LRU 策略淘汰最久未使用的项

分片缓存​

基本分片缓存​

分片缓存通过将数据分布在多个分片上减少锁竞争，提高并发性能：

// 创建一个32分片的内存缓存
shardedCache := cache.NewMemoryCache(cache.WithShardCount(32))

// 使用方式与普通缓存相同
ctx := context.Background()
shardedCache.Set(ctx, "key1", "value1", time.Hour)
value, err := shardedCache.Get(ctx, "key1")

高并发环境使用​

分片缓存在高并发环境中尤其有效：

// 高并发读写示例
func performHighConcurrentOperations(c cache.Cache) {
    var wg sync.WaitGroup
    ctx := context.Background()
    
    // 100个协程并发写入
    for i := 0; i < 100; i++ {
        wg.Add(1)
        go func(id int) {
            defer wg.Done()
            key := fmt.Sprintf("key:%d", id)
            c.Set(ctx, key, fmt.Sprintf("value:%d", id), time.Hour)
        }(i)
    }
    
    // 100个协程并发读取
    for i := 0; i < 100; i++ {
        wg.Add(1)
        go func(id int) {
            defer wg.Done()
            key := fmt.Sprintf("key:%d", id)
            c.Get(ctx, key)
        }(i)
    }
    
    wg.Wait()
}

基于节点的分片缓存​

NodeShardedCache 允许您在多个缓存节点之间分布数据，这是构建分布式缓存的基础：

// 创建基于节点的分片缓存
shardedCache := cache.NewNodeShardedCache()

// 创建三个独立的内存缓存作为节点
node1 := cache.NewMemoryCache()
node2 := cache.NewMemoryCache()
node3 := cache.NewMemoryCache()

// 将节点添加到分片缓存中，可以指定权重
shardedCache.AddNode("node1", node1, 1)  // 权重 1
shardedCache.AddNode("node2", node2, 1)  // 权重 1 
shardedCache.AddNode("node3", node3, 2)  // 权重 2，将接收约两倍的键

// 使用方式与普通缓存相同，但数据会基于键自动路由到正确的节点
ctx := context.Background()
shardedCache.Set(ctx, "user:1001", userData, time.Hour)

自定义哈希函数​

可以自定义一致性哈希使用的哈希函数：

// 自定义哈希函数
customHashFunc := func(data []byte) uint32 {
    h := fnv.New32a()
    h.Write(data)
    return h.Sum32()
}

// 创建使用自定义哈希函数的分片缓存
shardedCache := cache.NewNodeShardedCache(
    cache.WithNodeCustomHashFunc(customHashFunc),
)

设置副本因子​

副本因子控制每个数据项存储在多少个节点上，提高可用性：

// 创建副本因子为2的分片缓存（每个键存储在2个节点上）
shardedCache := cache.NewNodeShardedCache(
    cache.WithNodeReplicaFactor(2),
)

一致性策略​

FyerCache 提供多种缓存一致性策略，处理缓存与数据源的协调。

Cache-Aside 模式​

package main

import (
    "context"
    "fmt"
    "time"
    
    "github.com/fyerfyer/fyer-cache/cache"
    "github.com/fyerfyer/fyer-cache/cache/consistency"
)

// 简单的数据源实现
type UserDB struct {
    users map[string]string
}

func (db *UserDB) Load(ctx context.Context, key string) (any, error) {
    if val, exists := db.users[key]; exists {
        return val, nil
    }
    return nil, fmt.Errorf("user not found")
}

func (db *UserDB) Store(ctx context.Context, key string, value any) error {
    if str, ok := value.(string); ok {
        db.users[key] = str
        return nil
    }
    return fmt.Errorf("invalid value type")
}

func (db *UserDB) Remove(ctx context.Context, key string) error {
    delete(db.users, key)
    return nil
}

func main() {
    // 初始化缓存和数据源
    memCache := cache.NewMemoryCache()
    userDB := &UserDB{users: map[string]string{
        "user:1001": "John Doe",
        "user:1002": "Jane Smith",
    }}
    
    // 创建 Cache-Aside 策略
    cacheStrategy := consistency.NewCacheAsideStrategy(
        memCache, 
        userDB,
        consistency.WithCacheAsideTTL(10*time.Minute),
    )
    
    ctx := context.Background()
    
    // 使用策略获取数据，如果缓存未命中，会从数据源加载
    value, err := cacheStrategy.Get(ctx, "user:1001")
    if err != nil {
        fmt.Printf("Error: %v\n", err)
        return
    }
    fmt.Printf("User: %v\n", value)
    
    // 写入数据，会同时更新数据源和缓存
    err = cacheStrategy.Set(ctx, "user:1003", "Robert Brown", 10*time.Minute)
    if err != nil {
        fmt.Printf("Error: %v\n", err)
        return
    }
    
    // 删除数据，会同时从数据源和缓存中删除
    err = cacheStrategy.Del(ctx, "user:1002")
    if err != nil {
        fmt.Printf("Error: %v\n", err)
        return
    }
}

Write-Through 模式​

Write-Through 模式先更新数据源，再更新缓存：

// 创建 Write-Through 策略
writeThrough := consistency.NewWriteThroughStrategy(
    memCache,
    userDB,
    consistency.WithWriteThroughTTL(10*time.Minute),
    consistency.WithDelayedWrite(false, 0), // 不使用延迟写入
)

// 使用策略写入，会先写数据源，再写缓存
err := writeThrough.Set(ctx, "user:1004", "Alice Green", 10*time.Minute)

消息队列通知模式​

基于消息队列的缓存一致性模式适用于分布式环境：

// 简单实现一个消息发布者
type SimplePublisher struct {
    subscribers []func([]byte)
}

func (p *SimplePublisher) Publish(ctx context.Context, topic string, msg []byte) error {
    for _, sub := range p.subscribers {
        sub(msg)
    }
    return nil
}

func (p *SimplePublisher) Subscribe(topic string, handler func(msg []byte)) error {
    p.subscribers = append(p.subscribers, handler)
    return nil
}

func (p *SimplePublisher) Close() error {
    return nil
}

// 创建发布者
publisher := &SimplePublisher{}

// 创建基于消息队列的通知策略
mqStrategy := consistency.NewMQNotifierStrategy(
    memCache,
    userDB,
    publisher,
    consistency.WithMQTopic("cache:invalidate"),
    consistency.WithMQNotifierTTL(10*time.Minute),
)

// 使用策略，写入或删除操作会通过消息队列通知其他节点
err := mqStrategy.Set(ctx, "user:1005", "Tom Wilson", 10*time.Minute)

监控​

FyerCache 提供内置的监控功能，帮助您追踪缓存性能和使用情况。

基本监控​

package main

import (
    "time"

    "github.com/fyerfyer/fyer-cache/cache"
    "github.com/fyerfyer/fyer-cache/cache/metrics"
)

func main() {
    // 创建内存缓存
    memCache := cache.NewMemoryCache()
    
    // 创建带监控的缓存包装器
    monitoredCache := metrics.NewMonitoredCache(
        memCache,
        metrics.WithMetricsServer(":8080"),      // 在8080端口暴露指标
        metrics.WithNamespace("myapp"),          // 指标命名空间
        metrics.WithSubsystem("cache"),          // 指标子系统
        metrics.WithCollectInterval(10*time.Second), // 收集间隔
    )
    
    // 启动监控服务
    monitoredCache.Start()
    defer monitoredCache.Stop()
    
    // 使用monitoredCache代替memCache，所有操作都会被监控
    // monitoredCache.Get(), monitoredCache.Set(), monitoredCache.Del()
}

获取监控数据​

您可以通过 HTTP 获取 Prometheus 格式的监控数据：

curl http://localhost:8080/metrics

主要指标包括：

  • myapp_cache_hit_count - 缓存命中次数
  • myapp_cache_miss_count - 缓存未命中次数
  • myapp_cache_request_count - 按操作类型计数的请求数量
  • myapp_cache_operation_latency_seconds - 操作延迟分布
  • myapp_cache_memory_usage_bytes - 内存使用量
  • myapp_cache_item_count - 缓存项数量

命中率统计​

可以直接获取缓存命中率：

// 获取当前命中率
hitRate := monitoredCache.HitRate()
fmt.Printf("Current hit rate: %.2f%%\n", hitRate*100)

HTTP API服务器​

FyerCache 提供 HTTP API 服务器，方便通过 REST API 管理缓存：

package main

import (
    "time"
    
    "github.com/fyerfyer/fyer-cache/cache"
    "github.com/fyerfyer/fyer-cache/cache/api"
)

func main() {
    // 创建内存缓存
    memCache := cache.NewMemoryCache()
    
    // 创建 API 服务器
    apiServer := api.NewAPIServer(
        memCache,
        api.WithBindAddress(":8081"),     // 绑定地址
        api.WithBasePath("/api/cache"),   // API基础路径
        api.WithTimeouts(                 // 超时设置
            5*time.Second,  // 读超时
            5*time.Second,  // 写超时
            30*time.Second, // 空闲超时
        ),
    )
    
    // 启动 API 服务器
    if err := apiServer.Start(); err != nil {
        panic(err)
    }
    
    // 应用继续运行...
    // defer apiServer.Stop()
}

API 端点：

  • GET /api/cache/stats - 获取缓存统计信息
  • POST /api/cache/invalidate?key=X - 使指定的键失效

使用注意事项​

资源管理​

确保在应用程序退出前，正确关闭所有缓存资源：

// 内存缓存
memCache := cache.NewMemoryCache()
defer memCache.Close()

// 监控缓存
monitoredCache := metrics.NewMonitoredCache(memCache)
defer monitoredCache.Stop()

// API 服务器
apiServer := api.NewAPIServer(memCache)
defer apiServer.Stop()

错误处理​

处理缓存操作中可能出现的错误：

// 检查特定错误类型
val, err := cache.Get(ctx, "key")
if err != nil {
    if errors.Is(err, ferr.ErrKeyNotFound) {
        // 处理缓存未命中
    } else {
        // 处理其他错误
    }
}

并发考虑​

FyerCache 的所有实现都是并发安全的，但在高并发场景中：

  1. 使用足够的分片数减少锁竞争 (WithShardCount)
  2. 考虑使用异步清理 (WithAsyncCleanup) 避免清理过程阻塞缓存操作
  3. 对于分布式实现，设置合理的超时和重试策略
