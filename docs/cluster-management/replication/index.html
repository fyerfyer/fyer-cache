Data Replication

数据复制是 FyerCache 分布式系统中保证数据可靠性和高可用性的关键机制。

复制架构概述​

FyerCache 实现了基于领导者-跟随者（Leader-Follower）的数据复制架构，具有以下特点：

  1. 主从模式 - 一个主节点（Leader）处理写操作，多个从节点（Follower）提供读服务
  2. 复制日志 - 记录所有修改操作，用于数据同步和一致性保证
  3. 增量同步 - 正常情况下只传输新的变更
  4. 全量同步 - 新节点加入或长时间断连后进行完整数据同步
  5. 故障检测与恢复 - 自动检测节点故障并进行适当处理

架构组件关系如下：

┌────────────┐             ┌────────────┐
│            │             │            │
│   Leader   │──复制日志──▶│  Follower  │
│            │◀──心跳响应──│            │
└────────────┘             └────────────┘
      │                          │
      │                          │
      ▼                          ▼
┌────────────┐             ┌────────────┐
│            │             │            │
│  本地缓存   │             │  本地缓存   │
│            │             │            │
└────────────┘             └────────────┘

核心组件​

1. 复制日志（ReplicationLog）​

复制日志记录所有写操作，是数据复制的基础：

// ReplicationEntry 复制日志条目
type ReplicationEntry struct {
    Index      uint64        // 日志索引
    Term       uint64        // 任期号
    Command    string        // 命令类型 (Set, Del)
    Key        string        // 缓存键
    Value      []byte        // 缓存值
    Expiration time.Duration // 过期时间
    Timestamp  time.Time     // 操作时间戳
    ValueType  string        // 存储值类型
}

FyerCache 提供了一个基于内存的日志实现 MemoryReplicationLog，支持以下操作：

  • 追加日志：记录新的操作
  • 获取日志：按索引查询日志条目
  • 批量获取：获取一定范围的日志条目，用于同步
  • 截断日志：移除旧日志，节省内存空间
// NewMemoryReplicationLog 创建新的内存复制日志
log := replication.NewMemoryReplicationLog()

// 追加日志条目
entry := &replication.ReplicationEntry{
    Command: "Set",
    Key:     "user:1001",
    Value:   []byte(`{"name":"John","age":30}`),
    Term:    1,
}
log.Append(entry)

2. 同步器（Syncer）​

同步器负责在节点间传输和应用数据：

// Syncer 数据同步接口
type Syncer interface {
    // 记录设置操作到日志
    RecordSetOperation(key string, value []byte, expiration time.Duration, term uint64) error
    
    // 记录删除操作到日志
    RecordDelOperation(key string, term uint64) error
    
    // 启动同步器
    Start() error
    
    // 停止同步器
    Stop() error
    
    // 执行全量同步
    FullSync(ctx context.Context, target string) error
    
    // 执行增量同步
    IncrementalSync(ctx context.Context, target string, startIndex uint64) error
    
    // 应用同步数据
    ApplySync(ctx context.Context, entries []*ReplicationEntry) error
}

FyerCache 的默认实现 MemorySyncer 通过 HTTP 提供同步服务：

syncer := replication.NewMemorySyncer(
    "node-1",
    localCache,
    replicationLog,
    replication.WithAddress("localhost:8080"),
    replication.WithMaxRetries(3),
)
syncer.Start()

3. 领导者节点（LeaderNode）​

领导者节点负责处理写操作并将变更复制到跟随者：

// LeaderNode 领导者节点接口
type LeaderNode interface {
    // 获取/设置节点角色
    GetRole() ReplicationRole
    SetRole(role ReplicationRole)
    
    // 获取当前领导者ID
    GetLeader() string
    
    // 管理跟随者节点
    AddFollower(nodeID string, address string) error
    RemoveFollower(nodeID string) error
    GetFollowers() map[string]string
    
    // 启动和停止服务
    Start() error
    Stop() error
}

领导者节点维护从节点列表并处理心跳通信：

leader := replication.NewLeaderNode(
    "leader-1",
    localCache,
    replicationLog,
    syncer,
    replication.WithHeartbeatInterval(500 * time.Millisecond),
)
leader.Start()

// 添加从节点
leader.AddFollower("follower-1", "follower-1:8080")

4. 跟随者节点（FollowerNode）​

跟随者节点接收来自领导者的更新并应用到本地缓存：

// FollowerNode 跟随者节点接口
type FollowerNode interface {
    // 获取/设置节点角色
    GetRole() ReplicationRole
    SetRole(role ReplicationRole)
    
    // 获取/设置领导者
    GetLeader() string
    SetLeader(leaderID string, leaderAddr string)
    
    // 同步操作
    SyncFromLeader(ctx context.Context) error
    RequestFullSync(ctx context.Context) error
    
    // 启动和停止服务
    Start() error
    Stop() error
    
    // 升级为领导者
    PromoteToLeader() error
}

跟随者节点监控领导者心跳并处理复制请求：

follower := replication.NewFollowerNode(
    "follower-1",
    localCache,
    replicationLog,
    syncer,
    replication.WithElectionTimeout(2 * time.Second),
)
follower.Start()

// 设置领导者
follower.SetLeader("leader-1", "leader-1:8080")

// 从领导者同步数据
follower.SyncFromLeader(context.Background())

复制工作流程​

写操作流程​

当客户端发起写操作（Set/Del）时：

  1. 领导者处理请求：写操作必须发送到领导者节点
  2. 日志记录：领导者将操作记录到复制日志
  3. 应用本地：领导者将操作应用到自己的本地缓存
  4. 并行复制：领导者将日志条目并行发送给所有跟随者
  5. 确认完成：当大多数节点确认接收后，操作被视为持久化
// Leader 节点处理 Set 操作
func (l *LeaderNodeImpl) Set(ctx context.Context, key string, value interface{}, expiration time.Duration) error {
    // 1. 应用到本地缓存
    err := l.cache.Set(ctx, key, value, expiration)
    if err != nil {
        return err
    }
    
    // 2. 记录到复制日志并复制到从节点
    valueBytes, _ := json.Marshal(value)
    err = l.syncer.RecordSetOperation(key, valueBytes, expiration, l.currentTerm)
    if err != nil {
        return err
    }
    
    // 3. 触发复制到所有从节点
    go l.ReplicateEntries()
    
    return nil
}

读操作流程​

读操作可以由任何节点处理，但存在一致性差异：

  • 从领导者读取：保证最强一致性，但可能增加领导者负载
  • 从跟随者读取：提供更好的扩展性，但可能返回稍旧的数据（最终一致性）
// 从任意节点读取数据
func (n *ReplicationNode) Get(ctx context.Context, key string) (interface{}, error) {
    // 直接从本地缓存读取
    return n.cache.Get(ctx, key)
}

同步流程​

FyerCache 支持两种同步模式：

1. 增量同步​

正常情况下，跟随者只需获取自上次同步后的新日志条目：

// 跟随者执行增量同步
func (f *FollowerNodeImpl) SyncFromLeader(ctx context.Context) error {
    if f.leaderAddr == "" {
        return errors.New("no leader available")
    }
    
    // 获取本地最后日志索引
    lastIndex := f.log.GetLastIndex()
    
    // 从 Leader 增量同步数据
    return f.syncer.IncrementalSync(ctx, f.leaderAddr, lastIndex)
}

2. 全量同步​

在以下情况下执行全量同步：

  • 新节点加入集群
  • 节点长时间断连后重新连接
  • 日志不一致需要完全重建
// 跟随者请求全量同步
func (f *FollowerNodeImpl) RequestFullSync(ctx context.Context) error {
    if f.leaderAddr == "" {
        return errors.New("no leader available")
    }
    
    // 请求并应用完整数据
    return f.syncer.FullSync(ctx, f.leaderAddr)
}

故障检测与恢复​

心跳机制​

领导者定期向跟随者发送心跳，确认连接状态：

// 领导者发送心跳
func (l *LeaderNodeImpl) startHeartbeat() {
    ticker := time.NewTicker(l.config.HeartbeatInterval)
    defer ticker.Stop()
    
    for {
        select {
        case <-ticker.C:
            l.SendHeartbeat()
        case <-l.stopCh:
            return
        }
    }
}

跟随者监控心跳，检测领导者故障：

// 跟随者监控心跳
func (f *FollowerNodeImpl) startHeartbeatMonitor() {
    f.heartbeatTimer = time.AfterFunc(f.heartbeatTimeout, func() {
        // 检查最后一次心跳时间
        f.mu.RLock()
        elapsed := time.Since(f.lastHeartbeat)
        f.mu.RUnlock()
        
        // 如果超过超时时间未收到心跳，触发选举
        if elapsed > f.heartbeatTimeout {
            if f.onLeaderFail != nil {
                f.onLeaderFail()
            }
        }
    })
}

领导者故障处理​

当领导者故障时：

  1. 检测失效：跟随者检测到心跳超时
  2. 触发回调：调用配置的失效处理函数
  3. 选举流程：由集成方实现，通常基于优先级或外部协调
// 设置领导者失效回调
follower.SetElectionCallback(func() {
    // 这里可以实现选举逻辑或通知外部系统
    // 例如提升当前节点为领导者，或选择新领导者
    log.Printf("Leader %s appears to have failed", follower.GetLeader())
    
    // 在 LeaderFollowerCluster 中，会自动处理故障转移
    cluster.handleFailover()
})

主从集群集成​

LeaderFollowerCluster 类集成了主从复制和集群管理功能：

// 创建主从集群节点
cluster, err := integration.NewLeaderFollowerCluster(
    "node-1",
    "localhost:7946",  // 集群通信地址
    localCache,        // 本地缓存实例
    []cluster.NodeOption{
        cluster.WithGossipInterval(500 * time.Millisecond),
    },
    []integration.LeaderFollowerClusterOption{
        integration.WithSyncerAddress("localhost:8080"),  // 同步服务地址
        integration.WithInitialRole(replication.RoleLeader),  // 初始角色
    },
)

// 启动集群
cluster.Start()

// 第一个节点自引导
cluster.Join("localhost:7946")

// 其他节点加入集群
otherCluster.Join("localhost:7946")

主从集群提供自动故障转移功能：

// 自动处理领导者故障
func (lfc *LeaderFollowerCluster) handleFailover() {
    lfc.mu.Lock()
    defer lfc.mu.Unlock()
    
    // 检查当前角色，只有跟随者才能进行故障转移
    if lfc.currentRole != replication.RoleFollower {
        return
    }
    
    // 根据条件决定是否提升为领导者
    // 例如优先级、节点ID等
    shouldPromote := lfc.determineIfShouldPromote()
    
    if shouldPromote {
        lfc.PromoteToLeader()
    }
}

配置选项​

FyerCache 提供丰富的数据复制配置选项：

// 复制配置选项
type ReplicationOption func(*ReplicationConfig)

// 示例配置
leader := replication.NewLeaderNode(
    nodeID,
    cache,
    log,
    syncer,
    // 心跳间隔
    replication.WithHeartbeatInterval(500 * time.Millisecond),
    // 选举超时
    replication.WithElectionTimeout(2 * time.Second),
    // 最大日志条目数
    replication.WithMaxLogEntries(1000),
    // 快照阈值
    replication.WithSnapshotThreshold(10000),
    // 重试参数
    replication.WithMaxRetries(3),
    replication.WithRetryInterval(200 * time.Millisecond),
    // 复制超时
    replication.WithReplicationTimeout(5 * time.Second),
    // 选举优先级
    replication.WithPriority(10),
)

关键配置参数​

参数描述默认值建议值
HeartbeatInterval领导者发送心跳的间隔500ms200ms-1s，根据网络稳定性调整
ElectionTimeout跟随者检测领导者失效的超时时间1s1s-5s，通常为心跳间隔的2-4倍
MaxLogEntries单次同步最大日志条目数1000500-5000，根据条目大小调整
SnapshotThreshold触发快照的日志条目阈值100005000-50000，根据内存限制调整
MaxRetries操作最大重试次数32-5，根据网络稳定性调整
RetryInterval重试间隔200ms100ms-1s
ReplicationTimeout复制操作超时5s2s-10s，根据数据量和网络调整

实际应用示例​

基本主从复制设置​

package main

import (
    "context"
    "log"
    "time"
    
    "github.com/fyerfyer/fyer-cache/cache"
    "github.com/fyerfyer/fyer-cache/cache/replication"
)

func main() {
    // 创建缓存实例
    leaderCache := cache.NewMemoryCache()
    followerCache := cache.NewMemoryCache()
    
    // 创建复制日志
    leaderLog := replication.NewMemoryReplicationLog()
    followerLog := replication.NewMemoryReplicationLog()
    
    // 创建数据同步器
    leaderSyncer := replication.NewMemorySyncer("leader-1", leaderCache, leaderLog,
        replication.WithAddress("localhost:8080"))
    followerSyncer := replication.NewMemorySyncer("follower-1", followerCache, followerLog,
        replication.WithAddress("localhost:8081"))
    
    // 启动同步器
    leaderSyncer.Start()
    followerSyncer.Start()
    
    // 创建领导者节点
    leader := replication.NewLeaderNode("leader-1", leaderCache, leaderLog, leaderSyncer,
        replication.WithHeartbeatInterval(500 * time.Millisecond))
    
    // 创建跟随者节点
    follower := replication.NewFollowerNode("follower-1", followerCache, followerLog, followerSyncer,
        replication.WithElectionTimeout(2 * time.Second))
    
    // 启动节点
    leader.Start()
    follower.Start()
    
    // 配置跟随者指向领导者
    follower.SetLeader("leader-1", "localhost:8080")
    
    // 添加跟随者到领导者
    leader.AddFollower("follower-1", "localhost:8081")
    
    // 在领导者上写入数据
    ctx := context.Background()
    leaderCache.Set(ctx, "user:1001", "John Doe", 10*time.Minute)
    
    // 等待复制完成
    time.Sleep(1 * time.Second)
    
    // 从跟随者读取数据
    value, err := followerCache.Get(ctx, "user:1001")
    if err != nil {
        log.Printf("Error: %v", err)
    } else {
        log.Printf("Value from follower: %v", value)
    }
    
    // 清理资源
    leader.Stop()
    follower.Stop()
    leaderSyncer.Stop()
    followerSyncer.Stop()
}

集成主从复制和集群管理​

package main

import (
    "context"
    "log"
    "time"
    
    "github.com/fyerfyer/fyer-cache/cache"
    "github.com/fyerfyer/fyer-cache/cache/cluster"
    "github.com/fyerfyer/fyer-cache/cache/cluster/integration"
)

func main() {
    // 创建本地缓存
    cache1 := cache.NewMemoryCache()
    cache2 := cache.NewMemoryCache()
    cache3 := cache.NewMemoryCache()
    
    // 创建主从集群节点
    node1, err := integration.NewLeaderFollowerCluster(
        "node-1", 
        "localhost:7946", 
        cache1,
        []cluster.NodeOption{
            cluster.WithGossipInterval(500 * time.Millisecond),
        },
        []integration.LeaderFollowerClusterOption{
            integration.WithSyncerAddress("localhost:8080"),
            integration.WithInitialRole(replication.RoleLeader),
        },
    )
    if err != nil {
        log.Fatalf("Failed to create node1: %v", err)
    }
    
    node2, err := integration.NewLeaderFollowerCluster(
        "node-2", 
        "localhost:7947", 
        cache2,
        []cluster.NodeOption{
            cluster.WithGossipInterval(500 * time.Millisecond),
        },
        []integration.LeaderFollowerClusterOption{
            integration.WithSyncerAddress("localhost:8081"),
        },
    )
    if err != nil {
        log.Fatalf("Failed to create node2: %v", err)
    }
    
    node3, err := integration.NewLeaderFollowerCluster(
        "node-3", 
        "localhost:7948", 
        cache3,
        []cluster.NodeOption{
            cluster.WithGossipInterval(500 * time.Millisecond),
        },
        []integration.LeaderFollowerClusterOption{
            integration.WithSyncerAddress("localhost:8082"),
        },
    )
    if err != nil {
        log.Fatalf("Failed to create node3: %v", err)
    }
    
    // 启动所有节点
    node1.Start()
    node2.Start()
    node3.Start()
    
    // 确保正确清理
    defer func() {
        node1.Stop()
        node2.Stop()
        node3.Stop()
    }()
    
    // 形成集群
    node1.Join("localhost:7946")  // 第一个节点自引导
    time.Sleep(500 * time.Millisecond)
    
    node2.Join("localhost:7946")  // 其他节点加入第一个节点
    time.Sleep(500 * time.Millisecond)
    
    node3.Join("localhost:7946")
    
    // 等待集群稳定
    time.Sleep(2 * time.Second)
    
    // 写入数据到领导者
    ctx := context.Background()
    node1.Set(ctx, "key1", "value1", 10*time.Minute)
    
    // 等待复制完成
    time.Sleep(1 * time.Second)
    
    // 从跟随者节点读取数据
    value, err := node2.Get(ctx, "key1")
    if err != nil {
        log.Printf("Error reading from node2: %v", err)
    } else {
        log.Printf("Value from node2: %v", value)
    }
}
