Leader-Follower Architecture

FyerCache 提供了一种高可用的主从架构实现，允许您构建一个由一个主节点(Leader)和多个从节点(Follower)组成的缓存集群。

架构概述​

主从架构遵循以下原则：

  • 写入集中化：所有写操作(Set、Del)都由主节点处理，然后复制到从节点
  • 读取分布化：读操作(Get)可以由任何节点处理，提供负载均衡
  • 自动复制：主节点上的数据变更会自动复制到所有从节点
  • 故障转移：当主节点失效时，从节点可以被提升为新的主节点

架构组件关系如下：

┌────────────┐             ┌────────────┐
│            │             │            │
│   Leader   │──复制日志──▶│  Follower  │
│            │◀──心跳响应──│            │
└────────────┘             └────────────┘
      │                          │
      │                          │
      ▼                          ▼
┌────────────┐             ┌────────────┐
│            │             │            │
│  本地缓存   │             │  本地缓存   │
│            │             │            │
└────────────┘             └────────────┘

核心组件​

1. LeaderFollowerCluster​

LeaderFollowerCluster 是主从架构的集成点，它结合了集群管理和主从复制功能：

// LeaderFollowerCluster 实现了主从架构的缓存集群
type LeaderFollowerCluster struct {
    nodeID       string
    address      string
    localCache   cache.Cache
    clusterNode  *cluster.Node
    log          replication.ReplicationLog
    syncer       replication.Syncer
    leaderNode   replication.LeaderNode
    followerNode replication.FollowerNode
    currentRole  replication.ReplicationRole
    // ...其他字段
}

2. 主节点 (LeaderNode)​

主节点负责处理写操作并将变更复制到从节点：

// LeaderNode 领导者节点接口
type LeaderNode interface {
    // 获取/设置节点角色
    GetRole() ReplicationRole
    SetRole(role ReplicationRole)
    
    // 获取当前领导者ID
    GetLeader() string
    
    // 管理跟随者节点
    AddFollower(nodeID string, address string) error
    RemoveFollower(nodeID string) error
    GetFollowers() map[string]string
    
    // 启动和停止服务
    Start() error
    Stop() error
}

3. 从节点 (FollowerNode)​

从节点主要负责从主节点同步数据并处理读请求：

// FollowerNode 从节点接口
type FollowerNode interface {
    // 获取/设置节点角色
    GetRole() ReplicationRole
    SetRole(role ReplicationRole)
    
    // 获取/设置领导者
    GetLeader() string
    SetLeader(leaderID string, leaderAddr string)
    
    // 同步操作
    SyncFromLeader(ctx context.Context) error
    RequestFullSync(ctx context.Context) error
    
    // 启动和停止服务
    Start() error
    Stop() error
    
    // 升级为领导者
    PromoteToLeader() error
}

4. 复制日志 (ReplicationLog)​

复制日志记录所有修改操作，是实现数据一致性的关键：

// ReplicationLog 复制日志接口
type ReplicationLog interface {
    // 追加日志条目
    Append(entry *ReplicationEntry) error
    
    // 获取指定索引的日志
    Get(index uint64) (*ReplicationEntry, error)
    
    // 获取从指定索引开始的所有日志
    GetFrom(startIndex uint64, maxCount int) ([]*ReplicationEntry, error)
    
    // 获取最后一条日志的索引和任期
    GetLastIndex() uint64
    GetLastTerm() uint64
    
    // 截断日志
    Truncate(index uint64) error
}

5. 同步器 (Syncer)​

同步器处理节点间的数据传输，支持增量和全量同步：

// Syncer 数据同步接口
type Syncer interface {
    // 记录操作到日志
    RecordSetOperation(key string, value []byte, expiration time.Duration, term uint64) error
    RecordDelOperation(key string, term uint64) error
    
    // 同步方法
    FullSync(ctx context.Context, target string) error
    IncrementalSync(ctx context.Context, target string, startIndex uint64) error
    
    // 应用同步数据
    ApplySync(ctx context.Context, entries []*ReplicationEntry) error
    
    // 启动和停止服务
    Start() error
    Stop() error
}

主从复制工作流程​

写操作流程​

当客户端执行写操作(Set/Del)时：

  1. 客户端请求发送到任一节点
  2. 如果接收请求的是从节点，请求会转发给主节点
  3. 主节点执行以下步骤：
    • 将操作记录到复制日志
    • 应用操作到本地缓存
    • 异步将操作复制到所有从节点
  4. 返回操作结果给客户端
// Set 设置缓存值
func (lfc *LeaderFollowerCluster) Set(ctx context.Context, key string, val any, expiration time.Duration) error {
    lfc.mu.RLock()
    isLeader := lfc.currentRole == replication.RoleLeader
    lfc.mu.RUnlock()

    // 如果是领导者，直接处理请求
    if isLeader {
        // 应用到本地缓存
        if err := lfc.localCache.Set(ctx, key, val, expiration); err != nil {
            return err
        }
        
        // 记录到复制日志并触发复制
        valueBytes, _ := json.Marshal(val)
        return lfc.syncer.RecordSetOperation(key, valueBytes, expiration, lfc.getCurrentTerm())
    }
    
    // 如果不是领导者，转发请求
    leaderID := lfc.GetLeader()
    if leaderID == "" {
        return errors.New("no leader available")
    }
    
    // 调用领导者缓存
    // 实际实现中可能通过RPC或HTTP转发请求
    return errors.New("write not allowed on follower")
}

读操作流程​

当客户端执行读操作(Get)时：

  1. 客户端请求可以发送到任意节点
  2. 节点从本地缓存中查找数据
  3. 返回查询结果给客户端
// Get 从缓存获取值
func (lfc *LeaderFollowerCluster) Get(ctx context.Context, key string) (any, error) {
    // 无论角色，都直接从本地缓存读取
    return lfc.localCache.Get(ctx, key)
}

同步流程​

从节点通过两种方式与主节点保持同步：

1. 增量同步​

正常情况下，从节点只需要获取自上次同步后的新日志条目：

// 从节点执行增量同步
func (f *FollowerNodeImpl) SyncFromLeader(ctx context.Context) error {
    if f.leaderAddr == "" {
        return errors.New("no leader available")
    }
    
    // 获取本地最后日志索引
    lastIndex := f.log.GetLastIndex()
    
    // 从Leader增量同步数据
    return f.syncer.IncrementalSync(ctx, f.leaderAddr, lastIndex)
}

2. 全量同步​

在以下情况下执行全量同步：

  • 新节点加入集群
  • 节点长时间断连后重新连接
  • 日志不一致需要完全重建
// 从节点请求全量同步
func (f *FollowerNodeImpl) RequestFullSync(ctx context.Context) error {
    if f.leaderAddr == "" {
        return errors.New("no leader available")
    }
    
    // 请求并应用完整数据
    return f.syncer.FullSync(ctx, f.leaderAddr)
}

故障检测与恢复​

心跳机制​

主节点会定期向所有从节点发送心跳消息，用于保持连接和检测节点健康状态：

// 启动心跳定时器
func (l *LeaderNodeImpl) startHeartbeat() {
    if l.heartbeatTimer != nil {
        l.heartbeatTimer.Stop()
    }
    
    // 定期发送心跳到所有从节点
    l.heartbeatTimer = time.AfterFunc(l.config.HeartbeatInterval, func() {
        l.SendHeartbeat()
        l.startHeartbeat() // 重新安排下一次心跳
    })
}

从节点通过监控心跳来检测主节点是否存活：

// 启动心跳监控定时器
func (f *FollowerNodeImpl) startHeartbeatMonitor() {
    f.heartbeatTimer = time.AfterFunc(f.heartbeatTimeout, func() {
        f.mu.RLock()
        elapsed := time.Since(f.lastHeartbeat)
        f.mu.RUnlock()
        
        // 如果超过超时未收到心跳，触发选举
        if elapsed > f.heartbeatTimeout {
            if f.onLeaderFail != nil {
                f.onLeaderFail()
            }
        }
        
        // 重置定时器
        f.startHeartbeatMonitor()
    })
}

领导者故障处理​

当主节点失败时，系统通过以下流程选举新的主节点：

  1. 从节点检测到主节点心跳超时
  2. 触发故障转移回调
  3. 选择合适的从节点升级为新主节点
  4. 更新集群状态
// 设置选举回调函数
func (lfc *LeaderFollowerCluster) setElectionCallback() {
    if follower, ok := lfc.followerNode.(*replication.FollowerNodeImpl); ok {
        follower.SetElectionCallback(func() {
            // 发起选举
            lfc.handleFailover()
        })
    }
}

// 处理故障转移
func (lfc *LeaderFollowerCluster) handleFailover() {
    // 简化的选举实现：当前节点直接尝试成为领导者
    lfc.mu.Lock()
    defer lfc.mu.Unlock()
    
    // 如果已经是领导者，无需操作
    if lfc.currentRole == replication.RoleLeader {
        return
    }
    
    // 提升为领导者
    if err := lfc.PromoteToLeader(); err != nil {
        // 处理错误
        return
    }
    
    // 更新集群状态
}

配置选项​

通过配置选项可以自定义主从集群的行为：

// 创建主从集群节点
cluster, err := integration.NewLeaderFollowerCluster(
    "node-1",
    "localhost:7946",  // 集群通信地址
    localCache,        // 本地缓存实例
    integration.WithSyncerAddress("localhost:8080"),  // 同步服务地址
    integration.WithInitialRole(replication.RoleLeader),  // 初始角色
    integration.WithHeartbeatInterval(1*time.Second),  // 心跳间隔
    integration.WithElectionTimeout(3*time.Second),  // 选举超时
)

关键配置参数​

参数名说明默认值
InitialRole节点初始角色RoleFollower
HeartbeatInterval心跳发送间隔500ms
ElectionTimeout领导者选举超时1s
MaxRetries操作最大重试次数3
RetryInterval重试间隔200ms
ReplicationTimeout复制操作超时5s

使用示例​

创建和启动主从集群​

// 创建本地缓存
leaderCache := cache.NewMemoryCache()
followerCache := cache.NewMemoryCache()

// 创建主节点
leaderCluster, err := integration.NewLeaderFollowerCluster(
    "leader-node",
    "localhost:7946",
    leaderCache,
    integration.WithInitialRole(replication.RoleLeader),
    integration.WithSyncerAddress("localhost:8080"),
)
if err != nil {
    log.Fatalf("Failed to create leader: %v", err)
}

// 创建从节点
followerCluster, err := integration.NewLeaderFollowerCluster(
    "follower-node",
    "localhost:7947",
    followerCache,
    integration.WithInitialRole(replication.RoleFollower),
    integration.WithSyncerAddress("localhost:8081"),
)
if err != nil {
    log.Fatalf("Failed to create follower: %v", err)
}

// 启动节点
leaderCluster.Start()
followerCluster.Start()

// 确保优雅退出
defer func() {
    leaderCluster.Stop()
    followerCluster.Stop()
}()

// 主节点形成集群
leaderCluster.Join("localhost:7946")

// 从节点加入集群
followerCluster.Join("localhost:7946")

// 手动设置Leader
followerCluster.SetLeader("leader-node", "localhost:8080")

数据操作​

ctx := context.Background()

// 在主节点写入数据
err := leaderCluster.Set(ctx, "user:1001", "John Doe", 10*time.Minute)
if err != nil {
    log.Printf("Write error: %v", err)
}

// 等待复制完成
time.Sleep(500 * time.Millisecond)

// 从从节点读取数据
value, err := followerCluster.Get(ctx, "user:1001")
if err != nil {
    log.Printf("Read error: %v", err)
} else {
    log.Printf("Value from follower: %v", value)
}

手动故障转移​

// 模拟主节点故障
leaderCluster.Stop()

// 提升从节点为新的主节点
err = followerCluster.PromoteToLeader()
if err != nil {
    log.Printf("Promotion failed: %v", err)
} else {
    log.Println("Follower promoted to leader")
}

// 在新主节点上写入数据
err = followerCluster.Set(ctx, "new-key", "new-value", 10*time.Minute)
